363def parse_kraken_output_line(kraken_output_line: str) -> PatientData.BinData.KrakenOutputEntry:
364    """
365    Parses one line of Kraken Output files, where each line is a tab-delimited string with information on classification, sequence ID / Node name, taxonomic ID and scientific name, sequence length, and LCA / Lowest Common Ancestor mapping.
366
367    Parameters
368    ----------
369    kraken_output_line : str
370        One individual line passed from read_kraken_output that contains all Kraken Output information for its corresponding entry in the bin.
371    
372    Returns
373    -------
374    KrakenOutputEntry instance with properly typed information.
375    """
376    classified, seq_id, tax_id, seq_len, LCA_mapping = kraken_output_line.strip().split("\t") # split each line into these categories
377    classified = True if classified.lower() == 'c' else False # if the lower case version of a letter is equal to "c", classified is true 
378    seq_len = int(seq_len)
379    LCA_mapping = LCA_mapping.split(" ") # split the LCA mapping on every space so that every value is now a substring
380    return PatientData.BinData.KrakenOutputEntry(classified, seq_id, tax_id, seq_len, LCA_mapping)

401def read_kraken_report(kraken_report_filename: str) -> PatientData.BinData.KrakenReport:
402    """
403    Reads a whole KrakenReport file, line-by-line, creating a structure contained in the returned KrakenReport instance where top clades are stored and given children attributes to reference their sub-clades.
404
405    Parameters
406    ----------
407    kraken_report_filename : str
408        Filepath of Kraken Report file.
409    
410    Returns
411    -------
412    KrakenReport instance with top clades, and each of their children having appropriate references to their sub-clades.
413    """
414    with open(kraken_report_filename, 'r') as fs_in:
415        kraken_report_lines = fs_in.read().splitlines()
416    
417    kraken_report = PatientData.BinData.KrakenReport()
418    # Stack produced to keep track of parents and children needed to construct the KrakenReport tree-like structure.
419    clade_stack: list[PatientData.BinData.KrakenReportClade] = []
420
421    for kraken_report_line in kraken_report_lines:
422        perc_frag, num_frag_cov, num_frag_asgn, rank_code, num_tax_id, ind_sci_name = [field for field in kraken_report_line.split("\t")]
423        
424        # We directly convert properties that are numeric in value to numerical data types, and we have the option of
425        # stripping the scientific name to avoid indentation-produced whitespace.
426        perc_frag = float(perc_frag)
427        num_frag_cov = int(num_frag_cov)
428        num_frag_asgn = int(num_frag_asgn)
429        num_tax_id = int(num_tax_id)
430        
431        # Per line in the Kraken Report file, keep track of the KrakenReport clade we are currently processing.
432        current_kraken_report_clade = PatientData.BinData.KrakenReportClade(perc_frag, num_frag_cov, num_frag_asgn, rank_code, num_tax_id, ind_sci_name)
433        
434        # Add Kraken Report clades that are unindented, meaning they are root-level.
435        if current_kraken_report_clade.depth == 0:
436            kraken_report.top_clades.append(current_kraken_report_clade)
437        # If the clade stack is empty, initiate the clade stack with the first clade we encounter.
438        if not clade_stack:
439            clade_stack.append(current_kraken_report_clade)
440        else:
441            # If the depth is greater than the latest clade in the stack, we consider this current clade a child and append it to the stack.
442            if current_kraken_report_clade.depth > clade_stack[-1].depth:
443                clade_stack.append(current_kraken_report_clade)
444            else:
445                # If the depth is less than or equal to the latest clade in the stack, process the stack until this is not the case
446                while(current_kraken_report_clade.depth <= clade_stack[-1].depth):
447                    # If the depths of both the current clade and the latest clade are the same, this means a new sub-tree is up next for processing.
448                    # pop the latest clade and mark it as the child of its parent.
449                    if current_kraken_report_clade.depth == clade_stack[-1].depth:
450                        latest_clade = clade_stack.pop()
451                        # This is to avoid the root clades from being considered children, avoiding an IndexError.
452                        if len(clade_stack) > 0:
453                            clade_stack[-1].children.append(latest_clade)
454                        # Make sure to add the current clade, as this will be the start of the new branch of taxons we will consider and add to the stack.
455                        clade_stack.append(current_kraken_report_clade)
456                        # Make sure to break the statement as the loop needs to end once a new sub-tree is being addressed.
457                        break
458                    else:
459                        # If the depth of the current KrakenReport clade is less than the latest clade, 
460                        # keep popping and assigning the popped clades as children of the latest clade in the clade stack.
461                        # We want to do this till this sub-tree is fully processed and we can move on to the next sub-tree
462                        latest_clade = clade_stack.pop()
463                        clade_stack[-1].children.append(latest_clade)
464    # To process the final sub-tree, which is not addressed by the above while loop, 
465    # we repeat the popping and children update procedure done above until the clade stack is empty.
466    while(clade_stack):
467        latest_clade = clade_stack.pop()
468        # This is to avoid the root clades from being considered children, avoiding an IndexError.
469        if len(clade_stack) > 0:
470            clade_stack[-1].children.append(latest_clade)
471    
472    return kraken_report

402def read_kraken_report(kraken_report_filename: str) -> PatientData.BinData.KrakenReport:
403    """
404    Reads a whole KrakenReport file, line-by-line, creating a structure contained in the returned KrakenReport instance where top clades are stored and given children attributes to reference their sub-clades.
405
406    Parameters
407    ----------
408    kraken_report_filename : str
409        Filepath of Kraken Report file.
410    
411    Returns
412    -------
413    KrakenReport instance with top clades, and each of their children having appropriate references to their sub-clades.
414    """
415    with open(kraken_report_filename, 'r') as fs_in:
416        kraken_report_lines = fs_in.read().splitlines()
417    
418    kraken_report = PatientData.BinData.KrakenReport()
419    # Stack produced to keep track of parents and children needed to construct the KrakenReport tree-like structure.
420    clade_stack: list[PatientData.BinData.KrakenReportClade] = []
421
422    for kraken_report_line in kraken_report_lines:
423        perc_frag, num_frag_cov, num_frag_asgn, rank_code, num_tax_id, ind_sci_name = [field for field in kraken_report_line.split("\t")]
424        
425        # We directly convert properties that are numeric in value to numerical data types, and we have the option of
426        # stripping the scientific name to avoid indentation-produced whitespace.
427        perc_frag = float(perc_frag)
428        num_frag_cov = int(num_frag_cov)
429        num_frag_asgn = int(num_frag_asgn)
430        num_tax_id = int(num_tax_id)
431        
432        # Per line in the Kraken Report file, keep track of the KrakenReport clade we are currently processing.
433        current_kraken_report_clade = PatientData.BinData.KrakenReportClade(perc_frag, num_frag_cov, num_frag_asgn, rank_code, num_tax_id, ind_sci_name)
434        
435        # Add Kraken Report clades that are unindented, meaning they are root-level.
436        if current_kraken_report_clade.depth == 0:
437            kraken_report.top_clades.append(current_kraken_report_clade)
438        # If the clade stack is empty, initiate the clade stack with the first clade we encounter.
439        if not clade_stack:
440            clade_stack.append(current_kraken_report_clade)
441        else:
442            # If the depth is greater than the latest clade in the stack, we consider this current clade a child and append it to the stack.
443            if current_kraken_report_clade.depth > clade_stack[-1].depth:
444                clade_stack.append(current_kraken_report_clade)
445            else:
446                # If the depth is less than or equal to the latest clade in the stack, process the stack until this is not the case
447                while(current_kraken_report_clade.depth <= clade_stack[-1].depth):
448                    # If the depths of both the current clade and the latest clade are the same, this means a new sub-tree is up next for processing.
449                    # pop the latest clade and mark it as the child of its parent.
450                    if current_kraken_report_clade.depth == clade_stack[-1].depth:
451                        latest_clade = clade_stack.pop()
452                        # This is to avoid the root clades from being considered children, avoiding an IndexError.
453                        if len(clade_stack) > 0:
454                            clade_stack[-1].children.append(latest_clade)
455                        # Make sure to add the current clade, as this will be the start of the new branch of taxons we will consider and add to the stack.
456                        clade_stack.append(current_kraken_report_clade)
457                        # Make sure to break the statement as the loop needs to end once a new sub-tree is being addressed.
458                        break
459                    else:
460                        # If the depth of the current KrakenReport clade is less than the latest clade, 
461                        # keep popping and assigning the popped clades as children of the latest clade in the clade stack.
462                        # We want to do this till this sub-tree is fully processed and we can move on to the next sub-tree
463                        latest_clade = clade_stack.pop()
464                        clade_stack[-1].children.append(latest_clade)
465    # To process the final sub-tree, which is not addressed by the above while loop, 
466    # we repeat the popping and children update procedure done above until the clade stack is empty.
467    while(clade_stack):
468        latest_clade = clade_stack.pop()
469        # This is to avoid the root clades from being considered children, avoiding an IndexError.
470        if len(clade_stack) > 0:
471            clade_stack[-1].children.append(latest_clade)
472    
473    return kraken_report
